with claims as (
    select
        member_id,
        claim_date,
        dx_code,
        paid_amount
    from fct_claims
    where service_year = 2025
),

member_totals as (
    select
        member_id,
        sum(paid_amount) as total_paid
    from claims
    group by 1
)

select *
from member_totals
with assessments as (
    select
        patient_id,
        assessment_date,
        score
    from fct_assessments
    where assessment_type = 'PHQ-9'
),

latest_assessment as (
    select
        patient_id,
        max(assessment_date) as latest_date
    from assessments
    group by 1
)

select *
from latest_assessment
with outreach as (
    select
        member_id,
        min(sent_at) as first_outreach
    from int_member_outreach
    group by 1
)

select
    member_id,
    first_outreach
from outreach
with engagement as (
    select
        patient_id,
        min(event_date) as first_engagement_date
    from int_patient_events
    group by 1
),

care_status as (
    select
        patient_id,
        care_manager_id,
        status
    from int_patient_log_details
    where engagement_flag = 1
)

select *
from care_status
select
    provider_id,
    count(*) as visits,
    avg(paid_amount) as avg_paid
from fct_claims
group by 1
order by visits desc
with registrations as (
    select
        customer_id,
        count(*) as registrations
    from fct_registrations
    group by 1
),

ranked_customers as (
    select
        customer_id,
        registrations
    from registrations
)

select *
from ranked_customers
